427 static __hwloc_inline int
428 hwloc_plugin_check_namespace(const char *pluginname __hwloc_attribute_unused, const char *symbol __hwloc_attribute_unused)
429 {
430 #ifdef HWLOC_INSIDE_PLUGIN
431  void *sym;
432 #ifdef HWLOC_HAVE_LTDL
433  lt_dlhandle handle = lt_dlopen(NULL);
434 #else
435  void *handle = dlopen(NULL, RTLD_NOW|RTLD_LOCAL);
436 #endif
437  if (!handle)
438  /* cannot check, assume things will work */
439  return 0;
440 #ifdef HWLOC_HAVE_LTDL
441  sym = lt_dlsym(handle, symbol);
442  lt_dlclose(handle);
443 #else
444  sym = dlsym(handle, symbol);
445  dlclose(handle);
446 #endif
447  if (!sym) {
448  static int verboseenv_checked = 0;
449  static int verboseenv_value = 0;
450  if (!verboseenv_checked) {
451  const char *verboseenv = getenv("HWLOC_PLUGINS_VERBOSE");
452  verboseenv_value = verboseenv ? atoi(verboseenv) : 0;
453  verboseenv_checked = 1;
454  }
455  if (verboseenv_value)
456  fprintf(stderr, "Plugin `%s' disabling itself because it cannot find the `%s' core symbol.\n",
457  pluginname, symbol);
458  return -1;
459  }
460 #endif /* HWLOC_INSIDE_PLUGIN */
461  return 0;
462 }

480 static __hwloc_inline int
481 hwloc_filter_check_pcidev_subtype_important(unsigned classid)
482 {
483  unsigned baseclass = classid >> 8;
484  return (baseclass == 0x03 /* PCI_BASE_CLASS_DISPLAY */
485  || baseclass == 0x02 /* PCI_BASE_CLASS_NETWORK */
486  || baseclass == 0x01 /* PCI_BASE_CLASS_STORAGE */
487  || baseclass == 0x0b /* PCI_BASE_CLASS_PROCESSOR */
488  || classid == 0x0c04 /* PCI_CLASS_SERIAL_FIBER */
489  || classid == 0x0c06 /* PCI_CLASS_SERIAL_INFINIBAND */
490  || baseclass == 0x12 /* Processing Accelerators */);
491 }

522 static __hwloc_inline int
523 hwloc_filter_check_keep_object(hwloc_topology_t topology, hwloc_obj_t obj)
524 {
525  hwloc_obj_type_t type = obj->type;
526  enum hwloc_type_filter_e filter = HWLOC_TYPE_FILTER_KEEP_NONE;
527  hwloc_topology_get_type_filter(topology, type, &filter);
528  if (filter == HWLOC_TYPE_FILTER_KEEP_NONE)
529  return 0;
530  if (filter == HWLOC_TYPE_FILTER_KEEP_IMPORTANT) {
531  if (type == HWLOC_OBJ_PCI_DEVICE)
532  return hwloc_filter_check_pcidev_subtype_important(obj->attr->pcidev.class_id);
533  if (type == HWLOC_OBJ_OS_DEVICE)
534  return hwloc_filter_check_osdev_subtype_important(obj->attr->osdev.type);
535  }
536  return 1;
537 }
